  let movementNumber: number = 0;

    if (movements[i] === "(") {
      ++floor;
    } else {
      --floor;
    }
    if (floor === -1) {
      movementNumber = i;
      break;
    }
  }
  return movementNumber;
}

export function pairs(nums: string): number {
  const movements = nums.split("");
  let result: number = 0;
  for (let i = 0; i < movements.length - 1; i++) {
    if (movements[i] == movements[i + 1]) {
      result += parseInt(movements[i]);
    }
  }
  return result;
}
